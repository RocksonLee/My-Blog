1.1.1

Problem

Luogu

Solution

这一道题的思路就是把 A 类似这类的字母转化成 1 这类的数字 然后暴力加减即可

Code

#include <bits/stdc++.h>
using namespace std;

#define il inline
#define INF 0x3f3f3f3f
#define cl(a, b) memset(a, b, sizeof(a))
typedef long long ll;
typedef unsigned long long ull;

#define in(x) x = read()
il int read()
{
    int v = 1, x = 0;
    char ch = getchar();
    while (!isdigit(ch))
    {
        if (ch == '-') v = -1;
        ch = getchar();
    }
    while (isdigit(ch)) x = x * 10 + ch - '0', ch = getchar();
    return x * v;
}

#define N 100
char a[N], b[N];
int ans1 = 1, ans2 = 1, len;

int main()
{
    scanf("%s%s", a, b);
    len = strlen(a);
    for (int i = 0; i < len; i++) ans1 *= a[i] - 'A' + 1;
    len = strlen(b);
    for (int i = 0; i < len; i++) ans2 *= b[i] - 'A' + 1;
    if (ans1 % 47 == ans2 % 47)
        cout << "GO";
    else
        cout << "STAY";
    return 0;
}

这道题我在做匹配的时候直接使用string，string的包装及使用都非常的重要，很有用啊

匹配人，然后利用C++的整数运算会去掉小数部分的特性直接计算即可

Code

    {
        if (ch == '-') v = -1;
        ch = getchar();
    }
    while (isdigit(ch)) x = x * 10 + ch - '0', ch = getchar();
    return x * v;
}
map<string, int> mp;
int n, w, m;
string name[100], temp;
int main()
{
    cin >> n;
    for (int i = 0; i < n; i++) cin >> name[i];
    for (int i = 0; i < n; i++)
    {

            mp[temp] += w / m;
        }
    }
    for (int i = 0; i < n; i++) printf("%s %d\n", name[i].c_str(), mp[name[i]]);

1.1.3

Problem

Luogu

Solution

这道题其实挺简单的，直接加上当前月份的天数就可以到第二月的13日

1900.1.13是周六！

然后用一句很复杂（其实也不会）的三目运算符就可以解决当前月份的天数计算

(m == 2 ? (((y % 4 == 0 && y % 100 != 0) || y % 400 == 0) ? 29 : 28) : ((m == 1 || m == 3 || m == 5 || m == 7 || m == 8 || m == 10 || m == 12) ? 31 : 30))

先把最复杂的2月计算给分割开

然后2月分润不润

其他月分大不大

Code

#include <bits/stdc++.h>
using namespace std;

#define il inline
#define INF 0x3f3f3f3f
#define cl(a, b) memset(a, b, sizeof(a))
typedef long long ll;
typedef unsigned long long ull;

#define in(x) x = read()
il int read()
{
    int v = 1, x = 0;
    char ch = getchar();
    while (!isdigit(ch))
    {
        if (ch == '-') v = -1;
        ch = getchar();
    }
    while (isdigit(ch)) x = x * 10 + ch - '0', ch = getchar();
    return x * v;
}
int x, w, ans[10];
int main()
{
    in(x);
    w = 5;
    for (int y = 1900; y < 1900 + x; y++)
    {
        for (int m = 1; m <= 12; m++)
        {
            ans[w]++;
            w += (m == 2 ? (((y % 4 == 0 && y % 100 != 0) || y % 400 == 0) ? 29 : 28) : ((m == 1 || m == 3 || m == 5 || m == 7 || m == 8 || m == 10 || m == 12) ? 31 : 30));
            w %= 7;
        }
    }
    for (int i = 5; i < 12; i++) printf("%d ", ans[i % 7]);
    return 0;
}

1.1.4

Problem

Luogu

Solution

对于这一道题，我们应该尝试破环成链

然后对这个环跑一边，因为有白色是那种特殊的东西，所以我们特殊对待

白色可接前可接后

a 记录最接近的白色左边位置

b 记录串右边位置

c 记录当前颜色

最后记得 min(ans,n) 当你破环成链，然后整条链都是一种字符（不要抠字眼）时，那么会大于n，嗯

Code

#include <bits/stdc++.h>
using namespace std;

#define il inline
#define INF 0x3f3f3f3f
#define cl(a, b) memset(a, b, sizeof(a))
typedef long long ll;
typedef unsigned long long ull;

#define in(x) x = read()
il int read()
{
    int v = 1, x = 0;
    char ch = getchar();
    while (!isdigit(ch))
    {
        if (ch == '-') v = -1;
        ch = getchar();
    }
    while (isdigit(ch)) x = x * 10 + ch - '0', ch = getchar();
    return x * v;
}

#define N 1000
int n, a, b, ans, w;
char s[N], c;

int main()
{
    scanf("%d %s", &n, s);
    memcpy(s + n, s, n);
    for (int i = 0; i < n<<1; i++)
    {
        if (s[i] == 'w')
            b++, w++;
        else if (s[i] == c)
            b++, w = 0;
        else
            ans = max(ans, a + b), a = b - w, b = w + 1, w = 0, c = s[i];
    }
    ans=max(ans,a+b);
    printf("%d", min(ans,n));
    return 0;
}

Problem

Luogu

solution

对于这道题，我们直接在结构体编写函数，以及编写一个初始化函数

那么针对每一数，我们都可以搞一个答案出来

Code

#include <bits/stdc++.h>
using namespace std;

#define il inline
#define INF 0x3f3f3f3f
#define cl(a, b) memset(a, b, sizeof(a))
typedef long long ll;
typedef unsigned long long ull;

#define in(x) x = read()
il int read()
{
    int v = 1, x = 0;
    char ch = getchar();
    while (!isdigit(ch))
    {
        if (ch == '-') v = -1;
        ch = getchar();
    }
    while (isdigit(ch)) x = x * 10 + ch - '0', ch = getchar();
    return x * v;
}

struct Num
{
    int val[100], pp, n;
    Num(int x, int p)
    {
        for (n = 0; x; n++) val[n] = x % p, x /= p;
    }
    bool check()
    {
        for (int i = 0; i < n; i++)
            if (val[i] != val[n - i - 1]) return false;
        return true;
    }
};

il void out(Num qwq)
{
    for (int i = 0; i < qwq.n; i++)
    {
        printf("%d ", qwq.val[i]);
    }
    return;
}
int main()
{
    int ans = 0;
    int n, s;
    scanf("%d%d", &n, &s);
    for (int i = s + 1; ans < n; i++)
    {
        int ok = 0;
        for (int j = 2; j < 11; j++)
        {
            Num temp(i, j);
            if (temp.check()) ok++;
        }
        if (ok>=2) printf("%d\n", i), ans++;
    }
    return 0;
}

1.3.1

Problem

Luogu

Solution

我们思考一下，贪心，就是需要把最便宜的先买了再说

再买贵点的

所以排序一下，再去买

Code

#include <bits/stdc++.h>
using namespace std;

#define il inline
#define INF 0x3f3f3f3f
#define cl(a, b) memset(a, b, sizeof(a))
typedef long long ll;
typedef unsigned long long ull;

#define in(x) x = read()
il int read()
{
    int v = 1, x = 0;
    char ch = getchar();
    while (!isdigit(ch))
    {
        if (ch == '-') v = -1;
        ch = getchar();
    }
    while (isdigit(ch)) x = x * 10 + ch - '0', ch = getchar();
    return x * v;
}

#define N 200200
pair<int, int> q[N];
int x, n, ans;

int main()
{
    in(x), in(n);
    for (int i = 0; i < n; i++)
    {
        in(q[i].first);
        in(q[i].second);
    }
    sort(q, q + n);
    for (int i = 0; i < n && x > 0; i++)
    {
        ans += (x < q[i].second ? x : q[i].second) * q[i].first;
        x -= q[i].second;
    }
    printf("%d", ans);
    return 0;
}

1.3.2

Problem

Luogu

Solution

对于这道题，我们还是尝试贪心做法，先有一块木板从头铺到尾，然后再断开 最长 的地方，重复断开 m-1 处，就可以只使用 m 块木板了。

特判你的牛的数量如果比木板多，那就是不是直接放就好呢

Code

#include <bits/stdc++.h>
using namespace std;

#define il inline
#define INF 0x3f3f3f3f
#define cl(a, b) memset(a, b, sizeof(a))
typedef long long ll;
typedef unsigned long long ull;

#define in(x) x = read()
il int read()
{
    int v = 1, x = 0;
    char ch = getchar();
    while (!isdigit(ch))
    {
        if (ch == '-') v = -1;
        ch = getchar();
    }
    while (isdigit(ch)) x = x * 10 + ch - '0', ch = getchar();
    return x * v;
}
#define N 500
int m, s, c, a[N], w[N];
int cmp(int a, int b)
{
    return a > b;
}
int main()
{
    in(m), in(s), in(c);
    if (m > c) { printf("%d", c); return 0; }
    for (int i = 0; i < c; i++) in(a[i]);
    sort(a, a + c);
    int ans = a[c - 1] - a[0] + 1;
    for (int i = 1; i < c; i++) w[i] = a[i] - a[i - 1];
    sort(w + 1, w + c, cmp);
    for (int i = 1; i <= m - 1; i++) ans -= w[i] - 1;
    printf("%d\n", ans);
    return 0;
}

1.3.3

Problem

Luogu

Solution

暴力拆解每一位，我们从100-999直接循环就好了，用一个函数解决重复问题（值得学习）

Code

#include <bits/stdc++.h>
using namespace std;

#define il inline
#define INF 0x3f3f3f3f
#define cl(a, b) memset(a, b, sizeof(a))
typedef long long ll;
typedef unsigned long long ull;

#define in(x) x = read()
il int read()
{
    int v = 1, x = 0;
    char ch = getchar();
    while (!isdigit(ch))
    {
        if (ch == '-') v = -1;
        ch = getchar();
    }
    while (isdigit(ch)) x = x * 10 + ch - '0', ch = getchar();
    return x * v;
}

int a[20], n, ans;
il int check(int x)
{
    for (int i = 0; i < n; i++)
    {
        if (a[i] == x) return true;
    }
    return false;
}

il bool find(int xx, int p)
{
    for (int k = 0, x = xx; x; k++, x /= 10)
        if (k == p || !check(x % 10)) return false;
    return true;
}
int main()
{
    scanf("%d", &n);
    for (int i = 0; i < n; i++) scanf("%d", &a[i]);
    for (int i = 100; i <= 999; i++)
    {
        for (int j = 10; j <= 99; j++)
        {
            int now1 = i * (j % 10), now2 = i * (j / 10), now3 = i * j;
            if (find(i, 3) && find(j, 2) && find(now1, 3) && find(now2, 3) && find(now3, 4)) ans++;
        }
    }
    printf("%d", ans);
    return 0;
}
